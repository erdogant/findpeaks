Find peaks in high sampled dataset
^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^

.. code:: python

	# Load library
	import numpy as np
	from findpeaks import findpeaks

	# Data
	i = 10000
	xs = np.linspace(0,3.7*np.pi,i)
	X = (0.3*np.sin(xs) + np.sin(1.3 * xs) + 0.9 * np.sin(4.2 * xs) + 0.06 * np.random.randn(i))

	# Initialize
	fp = findpeaks(method='peakdetect')
	results = fp.fit(X)
	# Plot
	fp.plot1d()

	fp = findpeaks(method='topology', limit=1)
	results = fp.fit(X)
	fp.plot1d()
	fp.plot_persistence()

Find peaks with pre-processing
^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^


.. code:: python

	# Import library
	from findpeaks import findpeaks

	# Import example
	X = fp.import_example()

	# Initialize
	fp = findpeaks(method='topology', scale=True, denoise=10, togray=True, imsize=(50,100), verbose=3)

	# Fit
	results = fp.fit(X)

	# Plot all
	fp.plot()
	
	# Plot preprocessing
	fp.plot_preprocessing()
